<questions>
<!-- chaque question a la structure suivante -->
<question>
<auteur>Auteur 1</auteur>
<!-- facultatif -->
<auteur_mail>adresse mél de l'auteur</auteur_mail>
<!-- facultatif -->
<etablissement>Nom de l'tablissement</etablissement>
<!-- facultatif -->
<date_creation>date de création au format unix timestamp</date_creation>
<!-- facultatif -->
<date_modification>date de modification au format unix timestamp</date_modification>
<!-- facultatif -->
<famille>une_famille</famille>
<!-- famille proposée facultatif -->
<domaine>B1</domaine>
<alinea>2</alinea>
<titre>Texte de la question </titre>
<reponses>
<reponse>Texte de la réponse 1</reponse>

<document>nom du fichier avec extension (sans chemin) </document>

</documents>
</question>
<!-- autres questions avec la méme structure -->
</questions>
